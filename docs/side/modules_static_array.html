Static Array

Simple API over static array buffer allocated on stack.

Example

#import "std/static_array"

main :: fn () s32 {
	using std;

	arr: StaticArray(s32, 10);
	loop i := 0; i < 10; i += 1 {
		sarray_push(&arr, i);
	}

	sarray_erase(&arr, 1);
	sarray_erase(&arr, 4);

	loop i := 0; i < arr.len; i += 1 {
		print("%\n", arr.data[i]);
	}
	return 0;
}
